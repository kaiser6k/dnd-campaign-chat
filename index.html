function showConnFeedback(msg, type = 'info') { let fb = document.getElementById('sb-feedback'); if (!fb) { fb = document.createElement('div'); fb.id = 'sb-feedback'; fb.style.cssText = 'margin-top:var(--space-3);padding:var(--space-3);border-radius:var(--radius-md);font-size:var(--text-sm);white-space:pre-wrap;word-break:break-all'; const btn = document.querySelector('#tab-connect .btn.btn-primary'); btn.insertAdjacentElement('afterend', fb); } const colors = { info: { bg: 'rgba(200,150,10,0.12)', border: 'rgba(200,150,10,0.3)', text: 'var(--gold-bright)' }, success: { bg: 'rgba(42,107,58,0.2)', border: 'rgba(39,174,96,0.4)', text: 'var(--green-bright)' }, error: { bg: 'rgba(139,42,42,0.2)', border: 'rgba(192,57,43,0.4)', text: '#e07070' }, }; const c = colors[type] || colors.info; fb.style.background = c.bg; fb.style.border = `1px solid ${c.border}`; fb.style.color = c.text; fb.textContent = msg; } async function connectSupabase() { const url = document.getElementById('sb-url').value.trim(); const key = document.getElementById('sb-key').value.trim(); if (!url || !key) { showConnFeedback('⚠ Enter both Supabase URL and anon key.', 'error'); return; } if (!url.startsWith('https://')) { showConnFeedback('⚠ URL must start with https://', 'error'); return; } if (!key.startsWith('eyJ')) { showConnFeedback('⚠ That doesn\'t look like an anon key (should start with eyJ…). Make sure you\'re not using the service_role key.', 'error'); return; } showConnFeedback('⏳ Loading Supabase library…', 'info'); if (!window.supabase) { try { await new Promise((res, rej) => { const s = document.createElement('script'); s.src = 'https://cdn.jsdelivr.net/npm/@supabase/supabase-js@2/dist/umd/supabase.min.js'; s.onload = res; s.onerror = () => rej(new Error('Failed to load Supabase JS from CDN. Check your internet connection.')); document.head.appendChild(s); }); } catch (e) { showConnFeedback('✗ ' + e.message, 'error'); return; } } showConnFeedback('⏳ Creating client and testing connection…', 'info'); try { const client = window.supabase.createClient(url, key); // Probe: verify credentials + table exist const { data, error } = await client .from('campaign_messages') .select('id') .limit(1); if (error) { if (error.code === '42P01') { showConnFeedback('✗ Table not found.\n\nRun the SQL in the box below first, then try again.\n\nError: ' + error.message, 'error'); } else if (error.code === 'PGRST301' || error.message?.includes('JWT')) { showConnFeedback('✗ Invalid anon key.\n\nCopy the anon/public key from Supabase → Project Settings → API.\n\nError: ' + error.message, 'error'); } else { showConnFeedback('✗ Connection test failed.\n\nError code: ' + error.code + '\n' + error.message, 'error'); } return; } state.supabase = client; setConnStatus('connected', '● Live'); state.supabaseChannel = state.supabase .channel('room:' + state.roomCode) .on('postgres_changes', { event: 'INSERT', schema: 'public', table: 'campaign_messages', filter: `room=eq.${state.roomCode}` }, payload => { const row = payload.new; if (row.player_id === state.myPlayerId) return; addLocalMessage({ type: row.msg_type || 'chat', playerId: row.player_id, playerName: row.player_name, playerColor: row.player_color, playerRole: row.player_role, content: row.content, diceType: row.dice_type, diceResult: row.dice_result, }); }) .subscribe((status, err) => { if (status === 'SUBSCRIBED') { showConnFeedback('✓ Connected! Share room code with your party: ' + state.roomCode, 'success'); } else if (status === 'CHANNEL_ERROR' || status === 'TIMED_OUT') { showConnFeedback('✗ Realtime subscription failed: ' + (err?.message || status) + '\n\nCheck that you ran:\nalter publication supabase_realtime add table campaign_messages;', 'error'); } }); } catch (e) { showConnFeedback('✗ Unexpected error: ' + e.message, 'error'); } }
